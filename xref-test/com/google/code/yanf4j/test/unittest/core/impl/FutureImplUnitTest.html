35          public NotifyFutureRunner(FutureImpl<Boolean> future, long sleepTime, Throwable throwable) {
36              super();
37              this.future = future;
38              this.sleepTime = sleepTime;
39              this.throwable = throwable;
40          }
41

43          public void run() {
44              try {
45                  Thread.sleep(this.sleepTime);
46                  if (this.throwable != null) {
47                      this.future.failure(this.throwable);
48                  }
49                  else {
50                      this.future.setResult(true);
51                  }
52              }
53              catch (Exception e) {
54                  throw new RuntimeException(e);
55              }
56          }
57      }
