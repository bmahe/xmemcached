21  public class SimpleExample {

25  	public static void main(String[] args) {
26  		if (args.length < 1) {
27  			System.err.println("Useage:java SimpleExample [servers]");
28  			System.exit(1);
29  		}
30  		MemcachedClient memcachedClient = getMemcachedClient(args[0]);
31  		if (memcachedClient == null) {
32  			throw new NullPointerException(
33  					"Null MemcachedClient,please check memcached has been started");
34  		}
35  		try {
36  			// add a,b,c
37  			System.out.println("Add a,b,c");
38  			memcachedClient.set("a", 0, "Hello,xmemcached");
39  			memcachedClient.set("b", 0, "Hello,xmemcached");
40  			memcachedClient.set("c", 0, "Hello,xmemcached");
41  			// get a
42  			String value = memcachedClient.get("a");
43  			System.out.println("get a=" + value);
44  			System.out.println("delete a");
45  			// delete a
46  			memcachedClient.delete("a");
47  			// reget a
48  			value = memcachedClient.get("a");
49  			System.out.println("after delete,a=" + value);
50  
51  			System.out.println("Iterate all keys...");
52  			// iterate all keys
53  			KeyIterator it = memcachedClient.getKeyIterator(AddrUtil
54  					.getOneAddress(args[0]));
55  			while (it.hasNext()) {
56  				System.out.println(it.next());
57  			}
58  			System.out.println(memcachedClient.touch("b", 1000));
59  
60  		} catch (MemcachedException e) {
61  			log.error("MemcachedClient operation fail", e);
62  		} catch (TimeoutException e) {
63  			log.error("MemcachedClient operation timeout", e);
64  		} catch (InterruptedException e) {
65  			// ignore
66  			log.info(e.getMessage());
67  		}
68  		try {
69  			memcachedClient.shutdown();
70  		} catch (Exception e) {
71  			log.error("Shutdown MemcachedClient fail", e);
72  		}
73  	}
74  
75  	public static MemcachedClient getMemcachedClient(String servers) {
76  		try {
77  			// use text protocol by default
78  			MemcachedClientBuilder builder = new XMemcachedClientBuilder(
79  					AddrUtil.getAddresses(servers));
80  			return builder.build();
81  		} catch (IOException e) {
82  			log.error("Create MemcachedClient fail", e);
83  		}
