19  public class SimpleExample {
20  	public static void main(String[] args) {
21  		if (args.length < 1) {
22  			System.err.println("Useage:java SimpleExample [servers]");
23  			System.exit(1);
24  		}
25  		MemcachedClient memcachedClient = getMemcachedClient(args[0]);
26  		if (memcachedClient == null) {
27  			throw new NullPointerException(
28  					"Null MemcachedClient,please check memcached has been started");
29  		}
30  		try {
31  			// add a,b,c
32  			System.out.println("Add a,b,c");
33  			memcachedClient.set("a", 0, "Hello,xmemcached");
34  			memcachedClient.set("b", 0, "Hello,xmemcached");
35  			memcachedClient.set("c", 0, "Hello,xmemcached");
36  			// get a
37  			String value = memcachedClient.get("a");
38  			System.out.println("get a=" + value);
39  			System.out.println("delete a");
40  			// delete a
41  			memcachedClient.delete("a");
42  			// reget a
43  			value = memcachedClient.get("a");
44  			System.out.println("after delete,a=" + value);
45  
46  			System.out.println("Iterate all keys...");
47  			// iterate all keys
48  			KeyIterator it = memcachedClient.getKeyIterator(AddrUtil
49  					.getOneAddress(args[0]));
50  			while (it.hasNext()) {
51  				System.out.println(it.next());
52  			}
53  			System.out.println(memcachedClient.touch("b", 1000));
54  
55  		} catch (MemcachedException e) {
56  			System.err.println("MemcachedClient operation fail");
57  			e.printStackTrace();
58  		} catch (TimeoutException e) {
59  			System.err.println("MemcachedClient operation timeout");
60  			e.printStackTrace();
61  		} catch (InterruptedException e) {
62  			// ignore
63  		}
64  		try {
65  			memcachedClient.shutdown();
66  		} catch (Exception e) {
67  			System.err.println("Shutdown MemcachedClient fail");
68  			e.printStackTrace();
69  		}
70  	}

72  	public static MemcachedClient getMemcachedClient(String servers) {
73  		try {
74  			// use text protocol by default
75  			MemcachedClientBuilder builder = new XMemcachedClientBuilder(
76  					AddrUtil.getAddresses(servers));
77  			return builder.build();
78  		} catch (IOException e) {
79  			System.err.println("Create MemcachedClient fail");
80  			e.printStackTrace();
