26  public class SASLExample {
27  	public static void main(String[] args) {
28  		if (args.length < 3) {
29  			System.err
30  					.println("Useage:java SASLExample servers username password");
31  			System.exit(1);
32  		}
33  		MemcachedClient memcachedClient = getMemcachedClient(args[0], args[1],
34  				args[2]);
35  		if (memcachedClient == null) {
36  			throw new NullPointerException(
37  					"Null MemcachedClient,please check memcached has been started");
38  		}
39  		try {
40  			// add a
41  			System.out.println("Add a,b,c");
42  			memcachedClient.set("a", 0, "Hello,xmemcached");
43  			// get a
44  			String value = memcachedClient.get("a");
45  			System.out.println("get a=" + value);
46  			System.out.println("delete a");
47  			// delete a
48  			memcachedClient.delete("a");
49  			value = memcachedClient.get("a");
50  			System.out.println("after delete,a=" + value);
51  		} catch (MemcachedException e) {
52  			System.err.println("MemcachedClient operation fail");
53  			e.printStackTrace();
54  		} catch (TimeoutException e) {
55  			System.err.println("MemcachedClient operation timeout");
56  			e.printStackTrace();
57  		} catch (InterruptedException e) {
58  			// ignore
59  		}
60  		try {
61  			memcachedClient.shutdown();
62  		} catch (Exception e) {
63  			System.err.println("Shutdown MemcachedClient fail");
64  			e.printStackTrace();
65  		}
66  	}

68  	public static MemcachedClient getMemcachedClient(String servers,
69  			String username, String password) {
70  		try {
71  			MemcachedClientBuilder builder = new XMemcachedClientBuilder(
72  					AddrUtil.getAddresses(servers));
73  			builder.addAuthInfo(AddrUtil.getOneAddress(servers), AuthInfo
74  					.typical(username, password));
75  			// Must use binary protocol
76  			builder.setCommandFactory(new BinaryCommandFactory());
77  			return builder.build();
78  		} catch (IOException e) {
79  			System.err.println("Create MemcachedClient fail");
80  			e.printStackTrace();
81  		}
82  		return null;
83  	}
