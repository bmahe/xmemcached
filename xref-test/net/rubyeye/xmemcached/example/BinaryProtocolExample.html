25  	public static void main(String[] args) {
26  		if (args.length < 1) {
27  			System.err.println("Useage:java BinaryProtocolExample [servers]");
28  			System.exit(1);
29  		}
30  		MemcachedClient memcachedClient = getMemcachedClient(args[0]);
31  		if (memcachedClient == null) {
32  			throw new NullPointerException(
33  					"Null MemcachedClient,please check memcached has been started");
34  		}
35  		try {
36  			// add a
37  			System.out.println("Add a,b,c");
38  			memcachedClient.set("a", 0, "Hello,xmemcached");
39  			// get a
40  			String value = memcachedClient.get("a");
41  			System.out.println("get a=" + value);
42  			System.out.println("delete a");
43  			// delete a
44  			memcachedClient.delete("a");
45  			value = memcachedClient.get("a");
46  			System.out.println("after delete,a=" + value);
47  		} catch (MemcachedException e) {
48  			log.error("MemcachedClient operation fail", e);
49  		} catch (TimeoutException e) {
50  			log.error("MemcachedClient operation timeout", e);
51  		} catch (InterruptedException e) {
52  			// ignore
53  			log.info(e.getMessage());
54  		}
55  		try {
56  			memcachedClient.shutdown();
57  		} catch (Exception e) {
58  			log.error("Shutdown MemcachedClient fail", e);
59  		}
60  	}
61  
62  	public static MemcachedClient getMemcachedClient(String servers) {
63  		try {
64  			MemcachedClientBuilder builder = new XMemcachedClientBuilder(
65  					AddrUtil.getAddresses(servers));
66  			// use binary protocol
67  			builder.setCommandFactory(new BinaryCommandFactory());
68  			return builder.build();
69  		} catch (IOException e) {
70  			System.err.println("Create MemcachedClient fail");
71  			e.printStackTrace();
