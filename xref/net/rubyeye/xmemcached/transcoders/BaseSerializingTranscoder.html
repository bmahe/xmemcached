227 	private byte[] zipDecompress(byte[] in) {
228 		int size = in.length * COMPRESS_RATIO;
229 		ByteArrayInputStream bais = new ByteArrayInputStream(in);
230 		InflaterInputStream is = new InflaterInputStream(bais);
231 		ByteArrayOutputStream baos = new ByteArrayOutputStream(size);
232 		try {
233 			byte[] uncompressMessage = new byte[size];
234 			while (true) {
235 				int len = is.read(uncompressMessage);
236 				if (len <= 0) {
237 					break;
238 				}
239 				baos.write(uncompressMessage, 0, len);
240 			}
241 			baos.flush();
242 			return baos.toByteArray();
243 
244 		} catch (IOException e) {
245 			log.error("Failed to decompress data", e);
246 			//baos = null;
247 		} finally {
248 			try {
249 				is.close();
250 			} catch (IOException e) {
251 				log.error("failed to close InflaterInputStream");
252 			}
253 			try {
254 				bais.close();
255 			} catch (IOException e) {
256 				log.error("failed to close ByteArrayInputStream");
257 			}
258 			try {
259 				baos.close();
260 			} catch (IOException e) {
261 				log.error("failed to close ByteArrayOutputStream");
262 			}
263 		}
264 		return baos == null ? null : baos.toByteArray();
265 	}

267 	private byte[] gzipDecompress(byte[] in) {
268 		ByteArrayOutputStream bos = null;
269 		if (in != null) {
270 			ByteArrayInputStream bis = new ByteArrayInputStream(in);
271 			bos = new ByteArrayOutputStream();
272 			GZIPInputStream gis = null;
273 			try {
274 				gis = new GZIPInputStream(bis);
275 
276 				byte[] buf = new byte[64 * 1024];
277 				int r = -1;
278 				while ((r = gis.read(buf)) > 0) {
279 					bos.write(buf, 0, r);
280 				}
281 			} catch (IOException e) {
282 				log.error("Failed to decompress data", e);
283 				bos = null;
284 			} finally {
285 				if (gis != null) {
286 					try {
287 						gis.close();
288 					} catch (IOException e) {
289 						log.error("Close GZIPInputStream error", e);
290 					}
291 				}
292 				if (bis != null) {
293 					try {
294 						bis.close();
295 					} catch (IOException e) {
296 						log.error("Close ByteArrayInputStream error", e);
297 					}
298 				}
299 			}
300 		}
301 		return bos == null ? null : bos.toByteArray();
302 	}
