78  	public final Object decode(CachedData d) {
79  		Object obj = d.decodedObject;
80  		if (obj != null) {
81  			return obj;
82  		}
83  		byte[] data = d.data;
84  
85  		int flags = d.flag;
86  		if ((flags & COMPRESSED) != 0) {
87  			data = decompress(data);
88  		}
89  		flags = flags & SPECIAL_MASK;
90  		obj = decode0(d, data, flags);
91  		d.decodedObject = obj;
92  		return obj;
93  	}

95  	protected final Object decode0(CachedData cachedData,byte[] data, int flags) {
96  		Object rv = null;
97  		if ((cachedData.flag & SERIALIZED) != 0 && data != null) {
98  			rv = deserialize(data);
99  		} else {
100 			if (this.primitiveAsString) {
101 				if (flags == 0) {
102 					return decodeString(data);
103 				}
104 			}
105 			if (flags != 0 && data != null) {
106 				switch (flags) {
107 				case SPECIAL_BOOLEAN:
108 					rv = Boolean.valueOf(this.transcoderUtils
109 							.decodeBoolean(data));
110 					break;
111 				case SPECIAL_INT:
112 					rv = Integer.valueOf(this.transcoderUtils.decodeInt(data));
113 					break;
114 				case SPECIAL_LONG:
115 					rv = Long.valueOf(this.transcoderUtils.decodeLong(data));
116 					break;
117 				case SPECIAL_BYTE:
118 					rv = Byte.valueOf(this.transcoderUtils.decodeByte(data));
119 					break;
120 				case SPECIAL_FLOAT:
121 					rv = new Float(Float.intBitsToFloat(this.transcoderUtils
122 							.decodeInt(data)));
123 					break;
124 				case SPECIAL_DOUBLE:
125 					rv = new Double(Double
126 							.longBitsToDouble(this.transcoderUtils
127 									.decodeLong(data)));
128 					break;
129 				case SPECIAL_DATE:
130 					rv = new Date(this.transcoderUtils.decodeLong(data));
131 					break;
132 				case SPECIAL_BYTEARRAY:
133 					rv = data;
134 					break;
135 				default:
136 					log
137 							.warn(String.format("Undecodeable with flags %x",
138 									flags));
139 				}
140 			} else {
141 				rv = decodeString(data);
142 			}
143 		}
144 		return rv;
145 	}

152 	public final CachedData encode(Object o) {
153 		byte[] b = null;
154 		int flags = 0;
155 		if (o instanceof String) {
156 			b = encodeString((String) o);
157 		} else if (o instanceof Long) {
158 			if (this.primitiveAsString) {
159 				b = encodeString(o.toString());
160 			} else {
161 				b = this.transcoderUtils.encodeLong((Long) o);
162 			}
163 			flags |= SPECIAL_LONG;
164 		} else if (o instanceof Integer) {
165 			if (this.primitiveAsString) {
166 				b = encodeString(o.toString());
167 			} else {
168 				b = this.transcoderUtils.encodeInt((Integer) o);
169 			}
170 			flags |= SPECIAL_INT;
171 		} else if (o instanceof Boolean) {
172 			if (this.primitiveAsString) {
173 				b = encodeString(o.toString());
174 			} else {
175 				b = this.transcoderUtils.encodeBoolean((Boolean) o);
176 			}
177 			flags |= SPECIAL_BOOLEAN;
178 		} else if (o instanceof Date) {
179 			b = this.transcoderUtils.encodeLong(((Date) o).getTime());
180 			flags |= SPECIAL_DATE;
181 		} else if (o instanceof Byte) {
182 			if (this.primitiveAsString) {
183 				b = encodeString(o.toString());
184 			} else {
185 				b = this.transcoderUtils.encodeByte((Byte) o);
186 			}
187 			flags |= SPECIAL_BYTE;
188 		} else if (o instanceof Float) {
189 			if (this.primitiveAsString) {
190 				b = encodeString(o.toString());
191 			} else {
192 				b = this.transcoderUtils.encodeInt(Float
193 						.floatToRawIntBits((Float) o));
194 			}
195 			flags |= SPECIAL_FLOAT;
196 		} else if (o instanceof Double) {
197 			if (this.primitiveAsString) {
198 				b = encodeString(o.toString());
199 			} else {
200 				b = this.transcoderUtils.encodeLong(Double
201 						.doubleToRawLongBits((Double) o));
202 			}
203 			flags |= SPECIAL_DOUBLE;
204 		} else if (o instanceof byte[]) {
205 			b = (byte[]) o;
206 			flags |= SPECIAL_BYTEARRAY;
207 		} else {
208 			b = serialize(o);
209 			flags |= SERIALIZED;
210 		}
211 		assert b != null;
212 		if (this.primitiveAsString) {
213 			// It is not be SERIALIZED,so change it to string type
214 			if ((flags & SERIALIZED) == 0) {
215 				flags = 0;
216 			}
217 		}
218 		if (b.length > this.compressionThreshold) {
219 			byte[] compressed = compress(b);
220 			if (compressed.length < b.length) {
221 				if (log.isDebugEnabled()) {
222 					log.debug("Compressed " + o.getClass().getName() + " from "
223 							+ b.length + " to " + compressed.length);
224 				}
225 				b = compressed;
226 				flags |= COMPRESSED;
227 			} else {
228 				if (log.isDebugEnabled()) {
229 					log.debug("Compression increased the size of "
230 							+ o.getClass().getName() + " from " + b.length
231 							+ " to " + compressed.length);
232 				}
233 			}
234 		}
235 		return new CachedData(flags, b, this.maxSize, -1);
236 	}
