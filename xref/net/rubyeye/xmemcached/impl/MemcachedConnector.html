100 		public SessionMonitor() {
101 			this.setName("Heal-Session-Thread");
102 		}
103 
104 		@Override
105 		public void run() {
106 			while (MemcachedConnector.this.isStarted() && MemcachedConnector.this.enableHealSession) {
107 				ReconnectRequest request = null;
108 				try {
109 					request = MemcachedConnector.this.waitingQueue.take();
110 
111 					InetSocketAddress address = request
112 							.getInetSocketAddressWrapper()
113 							.getInetSocketAddress();
114 
115 					if (!MemcachedConnector.this.removedAddrSet
116 							.contains(address)) {
117 						boolean connected = false;
118 						Future<Boolean> future = MemcachedConnector.this
119 								.connect(request.getInetSocketAddressWrapper());
120 						request.setTries(request.getTries() + 1);
121 						try {
122 							log.warn("Trying to connect to "
123 									+ address.getAddress().getHostAddress()
124 									+ ":" + address.getPort() + " for "
125 									+ request.getTries() + " times");
126 							if (!future.get(
127 									MemcachedClient.DEFAULT_CONNECT_TIMEOUT,
128 									TimeUnit.MILLISECONDS)) {
129 								connected = false;
130 							} else {
131 								connected = true;
132 							}
133 						} catch (TimeoutException e) {
134 							future.cancel(true);
135 						} catch (ExecutionException e) {
136 							future.cancel(true);
137 						} finally {
138 							if (!connected) {
139 								this.rescheduleConnectRequest(request);
140 							} else {
141 								continue;
142 							}
143 						}
144 					} else {
145 						log.warn("Remove invalid reconnect task for " + address);
146 						// remove reconnect task
147 					}
148 				} catch (InterruptedException e) {
149 					// ignore,check status
150 				} catch (Exception e) {
151 					log.error("SessionMonitor connect error", e);
152 					this.rescheduleConnectRequest(request);
153 				}
154 			}
155 		}
156 
157 		private void rescheduleConnectRequest(ReconnectRequest request) {
158 			if (request == null) {
159 				return;
160 			}
161 			InetSocketAddress address = request.getInetSocketAddressWrapper()
162 					.getInetSocketAddress();
163 			// update timestamp for next reconnecting
164 			request.updateNextReconnectTimeStamp(MemcachedConnector.this.healSessionInterval
165 					* request.getTries());
166 			log.error("Reconnect to " + address.getAddress().getHostAddress()
167 					+ ":" + address.getPort() + " fail");
168 			// add to tail
169 			MemcachedConnector.this.waitingQueue.offer(request);
170 		}
171 	}

221 	public synchronized void addSession(Session session) {
222 		MemcachedSession tcpSession = (MemcachedSession) session;
223 
224 		InetSocketAddressWrapper addrWrapper = tcpSession
225 				.getInetSocketAddressWrapper();
226 		// Remember the first time address resolved and use it in all
227 		// application lifecycle.
228 		if (addrWrapper.getRemoteAddressStr() == null) {
229 			addrWrapper.setRemoteAddressStr(String.valueOf(session
230 					.getRemoteSocketAddress()));
231 		}
232 
233 		InetSocketAddress mainNodeAddress = addrWrapper.getMainNodeAddress();
234 		if (mainNodeAddress != null) {
235 			// It is a standby session
236 			this.addStandbySession(session, mainNodeAddress);
237 		} else {
238 			// It is a main session
239 			this.addMainSession(session);
240 			// Update main sessions
241 			this.updateSessions();
242 		}
243 	}

245 	private void addMainSession(Session session) {
246 		InetSocketAddress remoteSocketAddress = session
247 				.getRemoteSocketAddress();
248 		log.warn("Add a session: "
249 				+ SystemUtils.getRawAddress(remoteSocketAddress) + ":"
250 				+ remoteSocketAddress.getPort());
251 		Queue<Session> sessions = this.sessionMap.get(remoteSocketAddress);
252 		if (sessions == null) {
253 			sessions = new ConcurrentLinkedQueue<Session>();
254 			Queue<Session> oldSessions = this.sessionMap.putIfAbsent(
255 					remoteSocketAddress, sessions);
256 			if (null != oldSessions) {
257 				sessions = oldSessions;
258 			}
259 		}
260 		// If it is in failure mode,remove closed session from list
261 		if (this.failureMode) {
262 			Iterator<Session> it = sessions.iterator();
263 			while (it.hasNext()) {
264 				Session tmp = it.next();
265 				if (tmp.isClosed()) {
266 					it.remove();
267 					break;
268 				}
269 			}
270 		}
271 
272 		sessions.offer(session);
273 		// Remove old session and close it
274 		while (sessions.size() > this.connectionPoolSize) {
275 			Session oldSession = sessions.poll();
276 			((MemcachedSession) oldSession).setAllowReconnect(false);
277 			oldSession.close();
278 		}
279 	}

281 	private void addStandbySession(Session session,
282 			InetSocketAddress mainNodeAddress) {
283 		InetSocketAddress remoteSocketAddress = session
284 				.getRemoteSocketAddress();
285 		log.warn("Add a standby session: "
286 				+ SystemUtils.getRawAddress(remoteSocketAddress) + ":"
287 				+ remoteSocketAddress.getPort() + " for "
288 				+ SystemUtils.getRawAddress(mainNodeAddress) + ":"
289 				+ mainNodeAddress.getPort());
290 		List<Session> sessions = this.standbySessionMap.get(mainNodeAddress);
291 		if (sessions == null) {
292 			sessions = new CopyOnWriteArrayList<Session>();
293 			List<Session> oldSessions = this.standbySessionMap.putIfAbsent(
294 					mainNodeAddress, sessions);
295 			if (null != oldSessions) {
296 				sessions = oldSessions;
297 			}
298 		}
299 		sessions.add(session);
300 	}

312 	public void removeReconnectRequest(InetSocketAddress inetSocketAddress) {
313 		this.removedAddrSet.add(inetSocketAddress);
314 		Iterator<ReconnectRequest> it = this.waitingQueue.iterator();
315 		while (it.hasNext()) {
316 			ReconnectRequest request = it.next();
317 			if (request.getInetSocketAddressWrapper().getInetSocketAddress()
318 					.equals(inetSocketAddress)) {
319 				it.remove();
320 				log.warn("Remove invalid reconnect task for "
321 						+ request.getInetSocketAddressWrapper()
322 						.getInetSocketAddress());
323 			}
324 		}

340 	public synchronized void removeSession(Session session) {
341 		MemcachedTCPSession tcpSession = (MemcachedTCPSession) session;
342 		InetSocketAddressWrapper addrWrapper = tcpSession
343 				.getInetSocketAddressWrapper();
344 		InetSocketAddress mainNodeAddr = addrWrapper.getMainNodeAddress();
345 		if (mainNodeAddr != null) {
346 			this.removeStandbySession(session, mainNodeAddr);
347 		} else {
348 			this.removeMainSession(session);
349 		}

352 	private void removeMainSession(Session session) {
353 		InetSocketAddress remoteSocketAddress = session
354 				.getRemoteSocketAddress();
355 		// If it was in failure mode,we don't remove closed session from list.
356 		if (this.failureMode) {
357 			log.warn("Client in failure mode,we don't remove session "
358 					+ SystemUtils.getRawAddress(remoteSocketAddress) + ":"
359 					+ remoteSocketAddress.getPort());
360 			return;
361 		}
362 		log.warn("Remove a session: "
363 				+ SystemUtils.getRawAddress(remoteSocketAddress) + ":"
364 				+ remoteSocketAddress.getPort());
365 		Queue<Session> sessionQueue = this.sessionMap.get(session
366 				.getRemoteSocketAddress());
367 		if (null != sessionQueue) {
368 			sessionQueue.remove(session);
369 			if (sessionQueue.size() == 0) {
370 				this.sessionMap.remove(session.getRemoteSocketAddress());
371 			}
372 			this.updateSessions();
373 		}
374 	}

376 	private void removeStandbySession(Session session,
377 			InetSocketAddress mainNodeAddr) {
378 		List<Session> sessionList = this.standbySessionMap.get(mainNodeAddr);
379 		if (null != sessionList) {
380 			sessionList.remove(session);
381 			if (sessionList.size() == 0) {
382 				this.standbySessionMap.remove(mainNodeAddr);
383 			}
384 		}
385 	}

392 	@Override
393 	public void onConnect(SelectionKey key) throws IOException {
394 		key.interestOps(key.interestOps() & ~SelectionKey.OP_CONNECT);
395 		ConnectFuture future = (ConnectFuture) key.attachment();
396 		if (future == null || future.isCancelled()) {
397 			this.cancelKey(key);
398 			return;
399 		}
400 		try {
401 			if (!((SocketChannel) key.channel()).finishConnect()) {
402 				this.cancelKey(key);
403 				future.failure(new IOException("Connect to "
404 						+ SystemUtils.getRawAddress(future
405 								.getInetSocketAddressWrapper()
406 								.getInetSocketAddress())
407 								+ ":"
408 								+ future.getInetSocketAddressWrapper()
409 								.getInetSocketAddress().getPort() + " fail"));
410 			} else {
411 				key.attach(null);
412 				this.addSession(this.createSession(
413 						(SocketChannel) key.channel(),
414 						future.getInetSocketAddressWrapper()));
415 				future.setResult(Boolean.TRUE);
416 			}
417 		} catch (Exception e) {
418 			future.failure(e);
419 			this.cancelKey(key);
420 			throw new IOException("Connect to "
421 					+ SystemUtils.getRawAddress(future
422 							.getInetSocketAddressWrapper()
423 							.getInetSocketAddress())
424 							+ ":"
425 							+ future.getInetSocketAddressWrapper()
426 							.getInetSocketAddress().getPort() + " fail,"
427 							+ e.getMessage());
428 		}
429 	}

431 	private void cancelKey(SelectionKey key) throws IOException {
432 		try {
433 			if (key.channel() != null) {
434 				key.channel().close();
435 			}
436 		} finally {
437 			key.cancel();
438 		}

456 	public Future<Boolean> connect(InetSocketAddressWrapper addressWrapper)
457 			throws IOException {
458 		if (addressWrapper == null) {
459 			throw new NullPointerException("Null Address");
460 		}
461 		// Remove addr from removed set
462 		this.removedAddrSet.remove(addressWrapper.getInetSocketAddress());
463 		SocketChannel socketChannel = null;
464 		try {
465 			socketChannel = SocketChannel.open();
466 			this.configureSocketChannel(socketChannel);
467 			ConnectFuture future = new ConnectFuture(addressWrapper);
468 			if (!socketChannel.connect(addressWrapper.getInetSocketAddress())) {
469 				this.selectorManager.registerChannel(socketChannel,
470 						SelectionKey.OP_CONNECT, future);
471 			} else {
472 				this.addSession(this.createSession(socketChannel,
473 						addressWrapper));
474 				future.setResult(true);
475 			}
476 			return future;
477 		} catch (IOException e) {
478 			if (socketChannel != null) {
479 				socketChannel.close();
480 			}
481 			throw e;
482 		}
483 	}

491 	public Session send(final Command msg) throws MemcachedException {
492 		MemcachedSession session = (MemcachedSession) this.findSessionByKey(msg
493 				.getKey());
494 		if (session == null) {
495 			throw new MemcachedException(
496 					"There is no available connection at this moment");
497 		}
498 		// If session was closed,try to use standby memcached node
499 		if (session.isClosed()) {
500 			session = this.findStandbySession(session);
501 		}
502 		if (session.isClosed()) {
503 			throw new MemcachedException("Session("
504 					+ SystemUtils.getRawAddress(session
505 							.getRemoteSocketAddress()) + ":"
506 							+ session.getRemoteSocketAddress().getPort()
507 							+ ") has been closed");
508 		}
509 		if (session.isAuthFailed()) {
510 			throw new MemcachedException("Auth failed to connection "
511 					+ session.getRemoteSocketAddress());
512 		}
513 		session.write(msg);
514 		return session;
515 	}

517 	private MemcachedSession findStandbySession(MemcachedSession session) {
518 		if (this.failureMode) {
519 			List<Session> sessionList = this
520 					.getStandbySessionListByMainNodeAddr(session
521 							.getRemoteSocketAddress());
522 			if (sessionList != null && !sessionList.isEmpty()) {
523 				return (MemcachedTCPSession) sessionList.get(this.random
524 						.nextInt(sessionList.size()));
525 			}
526 		}
527 		return session;
528 	}

647 	public synchronized void quitAllSessions() {
648 		for (Session session : this.sessionSet) {
649 			((MemcachedSession) session).quit();
650 		}
651 		int sleepCount = 0;
652 		while (sleepCount++ < 5 && this.sessionSet.size() > 0) {
653 			try {
654 				this.wait(1000);
