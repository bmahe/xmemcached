27  /**
28   * ConnectionFactory instance that sets up a ketama compatible connection.
29   *
30   * <p>
31   * This implementation piggy-backs on the functionality of the
32   * <code>DefaultConnectionFactory</code> in terms of connections and queue
33   * handling. Where it differs is that it uses both the <code>
34   * KetamaNodeLocator</code>
35   * and the <code>HashAlgorithm.KETAMA_HASH</code> to provide consistent node
36   * hashing.
37   *
38   * @see http://www.last.fm/user/RJ/journal/2007/04/10/392555/
39   *
40   * </p>
41   */
42  /**
43   * Consistent Hash Algorithm implementation,based on TreeMap.tailMap(hash)
44   * method.

46   * @author dennis

92  	private final void buildMap(Collection<Session> list, HashAlgorithm alg) {
93  		TreeMap<Long, List<Session>> sessionMap = new TreeMap<Long, List<Session>>();
94  
95  		for (Session session : list) {
96  			String sockStr = null;
97  			if (this.cwNginxUpstreamConsistent) {
98  				InetSocketAddress serverAddress = session
99  						.getRemoteSocketAddress();
100 				sockStr = serverAddress.getAddress().getHostAddress();
101 				if (serverAddress.getPort() != DEFAULT_PORT) {
102 					sockStr = sockStr + ":" + serverAddress.getPort();
103 				}
104 			} else {
105 				if (session instanceof MemcachedTCPSession) {
106 					// Always use the first time resolved address.
107 					sockStr = ((MemcachedTCPSession) session)
108 							.getInetSocketAddressWrapper()
109 							.getRemoteAddressStr();
110 				}
111 				if (sockStr == null) {
112 					sockStr = String.valueOf(session.getRemoteSocketAddress());
113 				}
114 			}
115 			/**
116 			 * Duplicate 160 X weight references
117 			 */
118 			int numReps = NUM_REPS;
119 			if (session instanceof MemcachedTCPSession) {
120 				numReps *= ((MemcachedSession) session).getWeight();
121 			}
122 			if (alg == HashAlgorithm.KETAMA_HASH) {
123 				for (int i = 0; i < numReps / 4; i++) {
124 					byte[] digest = HashAlgorithm.computeMd5(sockStr + "-" + i);
125 					for (int h = 0; h < 4; h++) {
126 						long k = (long) (digest[3 + h * 4] & 0xFF) << 24
127 								| (long) (digest[2 + h * 4] & 0xFF) << 16
128 								| (long) (digest[1 + h * 4] & 0xFF) << 8
129 								| digest[h * 4] & 0xFF;
130 						this.getSessionList(sessionMap, k).add(session);
131 					}
132 
133 				}
134 			} else {
135 				for (int i = 0; i < numReps; i++) {
136 					long key = alg.hash(sockStr + "-" + i);
137 					this.getSessionList(sessionMap, key).add(session);
138 				}
139 			}
140 		}
141 		this.ketamaSessions = sessionMap;
142 		this.maxTries = list.size();
143 	}

145 	private List<Session> getSessionList(
146 			TreeMap<Long, List<Session>> sessionMap, long k) {
147 		List<Session> sessionList = sessionMap.get(k);
148 		if (sessionList == null) {
149 			sessionList = new ArrayList<Session>();
150 			sessionMap.put(k, sessionList);
151 		}
152 		return sessionList;
153 	}

155 	public final Session getSessionByKey(final String key) {
156 		if (this.ketamaSessions == null || this.ketamaSessions.size() == 0) {
157 			return null;
158 		}
159 		long hash = this.hashAlg.hash(key);
160 		Session rv = this.getSessionByHash(hash);
161 		int tries = 0;
162 		while (!this.failureMode && (rv == null || rv.isClosed())
163 				&& tries++ < this.maxTries) {
164 			hash = this.nextHash(hash, key, tries);
165 			rv = this.getSessionByHash(hash);
166 		}
167 		return rv;
168 	}

170 	public final Session getSessionByHash(final long hash) {
171 		TreeMap<Long, List<Session>> sessionMap = this.ketamaSessions;
172 		if (sessionMap.size() == 0) {
173 			return null;
174 		}
175 		Long resultHash = hash;
176 		if (!sessionMap.containsKey(hash)) {
177 			// Java 1.6 adds a ceilingKey method, but xmemcached is compatible
178 			// with jdk5,So use tailMap method to do this.
179 			SortedMap<Long, List<Session>> tailMap = sessionMap.tailMap(hash);
180 			if (tailMap.isEmpty()) {
181 				resultHash = sessionMap.firstKey();
182 			} else {
183 				resultHash = tailMap.firstKey();
184 			}
185 		}
186 		//
187 		// if (!sessionMap.containsKey(resultHash)) {
188 		// resultHash = sessionMap.ceilingKey(resultHash);
189 		// if (resultHash == null && sessionMap.size() > 0) {
190 		// resultHash = sessionMap.firstKey();
191 		// }
192 		// }
193 		List<Session> sessionList = sessionMap.get(resultHash);
194 		if (sessionList == null || sessionList.size() == 0) {
195 			return null;
196 		}
197 		int size = sessionList.size();
198 		return sessionList.get(this.random.nextInt(size));
199 	}

201 	public final long nextHash(long hashVal, String key, int tries) {
202 		long tmpKey = this.hashAlg.hash(tries + key);
203 		hashVal += (int) (tmpKey ^ tmpKey >>> 32);
204 		hashVal &= 0xffffffffL; /* truncate to 32-bits */
205 		return hashVal;
206 	}

208 	public final void updateSessions(final Collection<Session> list) {
209 		this.buildMap(list, this.hashAlg);
210 	}
