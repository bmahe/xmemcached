25  /**
26   * Session locator base on hash(key) mod sessions.size().Standard hash strategy
27   * 
28   * @author dennis
29   * 
30   */
31  public class ArrayMemcachedSessionLocator extends
32  		AbstractMemcachedSessionLocator {
33  
34  	private HashAlgorithm hashAlgorighm;
35  	private transient volatile List<List<Session>> sessions;
36  
37  	public ArrayMemcachedSessionLocator() {
38  		this.hashAlgorighm = HashAlgorithm.NATIVE_HASH;
39  	}
40  
41  	public ArrayMemcachedSessionLocator(HashAlgorithm hashAlgorighm) {
42  		this.hashAlgorighm = hashAlgorighm;
43  	}
44  
45  	public final void setHashAlgorighm(HashAlgorithm hashAlgorighm) {
46  		this.hashAlgorighm = hashAlgorighm;
47  	}
48  
49  	public final long getHash(int size, String key) {
50  		long hash = this.hashAlgorighm.hash(key);
51  		return hash % size;
52  	}
53  
54  	final Random rand = new Random();
55  
56  	public final Session getSessionByKey(final String key) {
57  		if (this.sessions == null || this.sessions.size() == 0) {
58  			return null;
59  		}
60  		// Copy on read
61  		List<List<Session>> sessionList = this.sessions;
62  		int size = sessionList.size();
63  		if (size == 0) {
64  			return null;
65  		}
66  		long start = this.getHash(size, key);
67  		List<Session> sessions = sessionList.get((int) start);
68  		Session session = getRandomSession(sessions);
69  
70  		// If it is not failure mode,get next available session
71  		if (!this.failureMode && (session == null || session.isClosed())) {
72  			long next = this.getNext(size, start);
73  			while ((session == null || session.isClosed()) && next != start) {
74  				sessions = sessionList.get((int) next);
75  				next = this.getNext(size, next);
76  				session = getRandomSession(sessions);
77  			}
78  		}
79  		return session;
80  	}
81  
82  	private Session getRandomSession(List<Session> sessions) {
83  		if (sessions == null || sessions.isEmpty())
84  			return null;
85  		return sessions.get(rand.nextInt(sessions.size()));
86  	}
87  
88  	public final long getNext(int size, long start) {
89  		if (start == size - 1) {
90  			return 0;
91  		} else {
92  			return start + 1;
93  		}
94  	}
95  
96  	public final void updateSessions(final Collection<Session> list) {
97  		if (list == null || list.isEmpty()) {
98  			this.sessions = Collections.emptyList();
99  			return;
100 		}
101 		Collection<Session> copySessions = list;
102 		List<List<Session>> tmpList = new ArrayList<List<Session>>();
103 		Session target = null;
104 		List<Session> subList = null;
105 		for (Session session : copySessions) {
106 			if (target == null) {
107 				target = session;
108 				subList = new ArrayList<Session>();
109 				subList.add(target);
110 			} else {
111 				if (session.getRemoteSocketAddress().equals(
112 						target.getRemoteSocketAddress())) {
113 					subList.add(session);
114 				} else {					
115 					tmpList.add(subList);
116 					target = session;
117 					subList = new ArrayList<Session>();
118 					subList.add(target);
119 				}
120 			}
121 		}
122 
123 		// The last one
124 		if (subList != null) {
125 			tmpList.add(subList);
126 		}
127 
128 		List<List<Session>> newSessions = new ArrayList<List<Session>>(
129 				tmpList.size() * 2);
130 		for (List<Session> sessions : tmpList) {
131 			if (sessions != null && !sessions.isEmpty()) {
132 				Session session = sessions.get(0);
133 				if (session instanceof MemcachedTCPSession) {
134 					int weight = ((MemcachedSession) session).getWeight();
135 					for (int i = 0; i < weight; i++) {
136 						newSessions.add(sessions);
137 					}
138 				} else {
139 					newSessions.add(sessions);
140 				}
141 			}
142 
143 		}
144 		this.sessions = newSessions;
145 	}
