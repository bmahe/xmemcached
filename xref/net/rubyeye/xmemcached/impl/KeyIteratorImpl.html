26  public final class KeyIteratorImpl implements KeyIterator {

28  	private LinkedList<String> currentKeyList;
29  	private final MemcachedClient memcachedClient;

31  	private long opTimeout = 1000;

35  			InetSocketAddress inetSocketAddress) {
36  		super();

38  		this.memcachedClient = memcachedClient;
39  		this.inetSocketAddress = inetSocketAddress;
40  	}

42  	public final InetSocketAddress getServerAddress() {
43  		return this.inetSocketAddress;
44  	}

47  		this.opTimeout = opTimeout;
48  	}

50  	public void close() {
51  		this.itemNumbersList.clear();
52  		this.currentKeyList.clear();
53  		this.currentKeyList = null;
54  	}

56  	public boolean hasNext() {
57  		return (this.itemNumbersList != null && !this.itemNumbersList.isEmpty())
58  				|| (this.currentKeyList != null && !this.currentKeyList
59  						.isEmpty());
60  	}
61  
62  	@SuppressWarnings("unchecked")
63  	public String next() throws MemcachedException, TimeoutException,
64  			InterruptedException {
65  		if (!hasNext()) {
66  			throw new NoSuchElementException();
67  		}
68  
69  		if (this.currentKeyList != null && !this.currentKeyList.isEmpty()) {
70  			return this.currentKeyList.remove();
71  		}
72  
73  		int itemNumber = this.itemNumbersList.remove();
74  		Queue<Session> sessions = this.memcachedClient.getConnector()
75  				.getSessionByAddress(this.inetSocketAddress);
76  		if (sessions == null || sessions.size() == 0) {
77  			throw new MemcachedException(
78  					"The memcached server is not connected,address="
79  							+ this.inetSocketAddress);
80  		}
81  		Session session = sessions.peek();
82  		CountDownLatch latch = new CountDownLatch(1);
83  		if (this.memcachedClient.getProtocol() == Protocol.Text) {
84  			TextCacheDumpCommand textCacheDumpCommand = new TextCacheDumpCommand(
85  					latch, itemNumber);
86  			session.write(textCacheDumpCommand);
87  			if (!latch.await(this.opTimeout, TimeUnit.MILLISECONDS)) {
88  				throw new TimeoutException("stats cachedump timeout");
89  			}
90  			if (textCacheDumpCommand.getException() != null) {
91  				if (textCacheDumpCommand.getException() instanceof MemcachedException) {
92  					throw (MemcachedException) textCacheDumpCommand
93  							.getException();
94  				} else {
95  					throw new MemcachedServerException(textCacheDumpCommand
96  							.getException());
97  				}
98  			}
99  			this.currentKeyList = (LinkedList<String>) textCacheDumpCommand
100 					.getResult();
101 		} else {
102 			throw new MemcachedException(
103 					this.memcachedClient.getProtocol().name()
104 							+ " protocol doesn't support iterating all keys in memcached");
105 		}
106 		return next();
107 	}
