23  package net.rubyeye.xmemcached.command.text;

25  import java.nio.ByteBuffer;
26  import java.util.concurrent.CountDownLatch;
27  
28  import net.rubyeye.xmemcached.command.Command;
29  import net.rubyeye.xmemcached.command.CommandType;
30  import net.rubyeye.xmemcached.impl.MemcachedTCPSession;
31  import net.rubyeye.xmemcached.monitor.Constants;
32  import net.rubyeye.xmemcached.utils.ByteUtils;

34  import com.google.code.yanf4j.buffer.IoBuffer;

36  /**
37   * FlushAll command for text protocol
38   * 
39   * @author dennis
40   * 
41   */
42  public class TextFlushAllCommand extends Command {
43  
44  	public static final ByteBuffer FLUSH_ALL = ByteBuffer.wrap("flush_all\r\n"
45  			.getBytes());
46  
47  	protected int exptime;
48  
49  	public final int getExptime() {
50  		return this.exptime;
51  	}
52  
53  	public TextFlushAllCommand(final CountDownLatch latch, int delay,
54  			boolean noreply) {
55  		super("[flush_all]", (byte[]) null, latch);
56  		this.commandType = CommandType.FLUSH_ALL;
57  		this.exptime = delay;
58  		this.noreply = noreply;
59  	}
60  
61  	@Override
62  	public boolean decode(MemcachedTCPSession session, ByteBuffer buffer) {
63  		if (buffer == null || !buffer.hasRemaining()) {
64  			return false;
65  		}
66  		if (this.result == null) {
67  			byte first = buffer.get(buffer.position());
68  			if (first == 'O') {
69  				setResult(Boolean.TRUE);
70  				countDownLatch();
71  				// OK\r\n
72  				return ByteUtils.stepBuffer(buffer, 4);
73  			} else {
74  				return decodeError(session, buffer);
75  			}
76  		} else {
77  			return ByteUtils.stepBuffer(buffer, 4);
78  		}
79  	}
80  
81  	@Override
82  	public final void encode() {
83  		if (isNoreply()) {
84  			if (this.exptime <= 0) {
85  				this.ioBuffer = IoBuffer.allocate("flush_all".length() + 1
86  						+ Constants.NO_REPLY.length + 2);
87  				ByteUtils.setArguments(this.ioBuffer, "flush_all",
88  						Constants.NO_REPLY);
89  			} else {
90  				byte[] delayBytes = ByteUtils.getBytes(String
91  						.valueOf(this.exptime));
92  				this.ioBuffer = IoBuffer.allocate("flush_all".length() + 2
93  						+ delayBytes.length + Constants.NO_REPLY.length + 2);
94  				ByteUtils.setArguments(this.ioBuffer, "flush_all", delayBytes,
95  						Constants.NO_REPLY);
96  			}
97  			this.ioBuffer.flip();
98  		} else {
99  			if (this.exptime <= 0) {
100 				this.ioBuffer = IoBuffer.wrap(FLUSH_ALL.slice());
101 			} else {
102 
103 				byte[] delayBytes = ByteUtils.getBytes(String
104 						.valueOf(this.exptime));
105 				this.ioBuffer = IoBuffer.allocate("flush_all".length() + 1
106 						+ delayBytes.length + 2);
107 				ByteUtils.setArguments(this.ioBuffer, "flush_all", delayBytes);
108 			}
109 		}
110 	}
