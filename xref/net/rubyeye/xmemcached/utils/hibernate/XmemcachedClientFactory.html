43  /**
44   * Parses hibernate properties to produce a MemcachedClient.<br/>
45   * See {@link com.googlecode.hibernate.memcached.MemcachedCacheProvider} for
46   * property details. SPI for xmemcached.
47   * 
48   * @author dennis
49   */
50  public class XmemcachedClientFactory implements MemcacheClientFactory {
51  
52  	public static final String PROP_SERVERS = Config.PROP_PREFIX + "servers";
53  	public static final String PROP_READ_BUFFER_SIZE = Config.PROP_PREFIX
54  			+ "readBufferSize";
55  	public static final String PROP_OPERATION_TIMEOUT = Config.PROP_PREFIX
56  			+ "operationTimeout";
57  	public static final String PROP_HASH_ALGORITHM = Config.PROP_PREFIX
58  			+ "hashAlgorithm";
59  	public static final String PROP_COMMAND_FACTORY = Config.PROP_PREFIX
60  			+ "commandFactory";
61  
62  	public static final String PROP_SESSION_LOCATOR = Config.PROP_PREFIX
63  			+ "sessionLocator";
64  	
65  	public static final String PROP_CONNECTION_POOL_SIZE = Config.PROP_PREFIX
66              + "connectionPoolSize";
67  	
68  	public static final String PROP_CONNECT_TIMEOUT = Config.PROP_PREFIX
69              + "connectTimeout";
70  
71  	private final PropertiesHelper properties;
72  
73  	public XmemcachedClientFactory(PropertiesHelper properties) {
74  		this.properties = properties;
75  	}
76  
77  	public Memcache createMemcacheClient() throws Exception {
78  		MemcachedClientBuilder builder = new XMemcachedClientBuilder(AddrUtil
79  				.getAddresses(getServerList()));
80  		builder.setCommandFactory(getCommandFactory());
81  		builder.setSessionLocator(getSessionLocator());
82  		builder.getConfiguration()
83  				.setSessionReadBufferSize(getReadBufferSize());
84  		builder.setConnectionPoolSize(getConnectionPoolSize());
85  		builder.setConnectTimeout(getConnectTimeoutMillis());
86  		MemcachedClient client = builder.build();
87  		client.setOpTimeout(getOperationTimeoutMillis());
88  		return new Xmemcache(client);
89  	}
90  
91  	protected MemcachedSessionLocator getSessionLocator() {
92  		if (sessionLocatorNameEquals(ArrayMemcachedSessionLocator.class)) {
93  			return new ArrayMemcachedSessionLocator(getHashAlgorithm());
94  		}
95  
96  		if (sessionLocatorNameEquals(KetamaMemcachedSessionLocator.class)) {
97  			return new KetamaMemcachedSessionLocator(getHashAlgorithm());
98  		}
99  
100 		if (sessionLocatorNameEquals(PHPMemcacheSessionLocator.class)) {
101 			return new PHPMemcacheSessionLocator(getHashAlgorithm());
102 		}
103 
104 		throw new IllegalArgumentException("Unsupported "
105 				+ PROP_SESSION_LOCATOR + " value: " + getCommandFactoryName());
106 	}
107 
108 	protected CommandFactory getCommandFactory() {
109 		if (commandFactoryNameEquals(TextCommandFactory.class)) {
110 			return new TextCommandFactory();
111 		}
112 
113 		if (commandFactoryNameEquals(BinaryCommandFactory.class)) {
114 			return new BinaryCommandFactory();
115 		}
116 
117 		throw new IllegalArgumentException("Unsupported "
118 				+ PROP_COMMAND_FACTORY + " value: " + getCommandFactoryName());
119 	}
120 
121 	private boolean commandFactoryNameEquals(Class<?> cls) {
122 		return cls.getSimpleName().equals(getCommandFactoryName());
123 	}
124 	
125 	private boolean sessionLocatorNameEquals(Class<?> cls) {
126 		return cls.getSimpleName().equals(getSessionLocatorName());
127 	}
128 
129 	public String getServerList() {
130 		return this.properties.get(PROP_SERVERS, "localhost:11211");
131 	}
132 
133 	public int getReadBufferSize() {
134 		return this.properties.getInt(PROP_READ_BUFFER_SIZE,
135 				MemcachedClient.DEFAULT_SESSION_READ_BUFF_SIZE);
136 	}
137 	
138 	
139     public int getConnectionPoolSize() {
140         return this.properties.getInt(PROP_CONNECTION_POOL_SIZE,
141                 MemcachedClient.DEFAULT_CONNECTION_POOL_SIZE);
142     }
143 
144 	public long getOperationTimeoutMillis() {
145 		return this.properties.getLong(PROP_OPERATION_TIMEOUT,
146 				MemcachedClient.DEFAULT_OP_TIMEOUT);
147 	}
148 
149 	public long getConnectTimeoutMillis() {
150         return this.properties.getLong(PROP_CONNECT_TIMEOUT,
151                 MemcachedClient.DEFAULT_CONNECT_TIMEOUT);
152     }
153 	
154 	public HashAlgorithm getHashAlgorithm() {
155 		return this.properties.getEnum(PROP_HASH_ALGORITHM,
156 				HashAlgorithm.class, HashAlgorithm.NATIVE_HASH);
157 	}
158 
159 	public String getCommandFactoryName() {
160 		return this.properties.get(PROP_COMMAND_FACTORY,
161 				TextCommandFactory.class.getSimpleName());
162 	}
163 
164 	public String getSessionLocatorName() {
165 		return this.properties.get(PROP_SESSION_LOCATOR,
166 				ArrayMemcachedSessionLocator.class.getSimpleName());
167 	}
168 
169 	protected PropertiesHelper getProperties() {
170 		return this.properties;
171 	}
