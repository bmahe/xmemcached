229 	public Object getObject() throws Exception {
230 		this.checkAttribute();
231 		Map<InetSocketAddress, InetSocketAddress> serverMap = this
232 				.getServerMap();
233 		int[] weightsArray = this.getWeightsArray(serverMap);
234 		MemcachedClientBuilder builder = this.newBuilder(serverMap,
235 				weightsArray);
236 		this.configBuilder(builder);
237 		this.memcachedClient = builder.build();
238 		this.memcachedClient.setOpTimeout(opTimeout);
239 		return this.memcachedClient;
240 	}

242 	private MemcachedClientBuilder newBuilder(
243 			Map<InetSocketAddress, InetSocketAddress> serverMap,
244 			int[] weightsArray) {
245 		MemcachedClientBuilder builder;
246 		if (weightsArray == null) {
247 			builder = new XMemcachedClientBuilder(serverMap);
248 		} else {
249 			builder = new XMemcachedClientBuilder(serverMap, weightsArray);
250 		}
251 		return builder;
252 	}

272 	private int[] getWeightsArray(
273 			Map<InetSocketAddress, InetSocketAddress> serverMap) {
274 		int[] weightsArray = null;
275 		if (serverMap != null && serverMap.size() > 0 && this.weights != null) {
276 			if (this.weights.size() < serverMap.size()) {
277 				throw new IllegalArgumentException(
278 						"Weight list's size is less than server list's size");
279 			}
280 			weightsArray = new int[this.weights.size()];
281 			for (int i = 0; i < weightsArray.length; i++) {
282 				weightsArray[i] = this.weights.get(i);
283 			}
284 		}
285 		return weightsArray;
286 	}

298 	private void checkAttribute() {
299 		if (this.bufferAllocator == null) {
300 			throw new NullPointerException("Null BufferAllocator");
301 		}
302 		if (this.sessionLocator == null) {
303 			throw new NullPointerException("Null MemcachedSessionLocator");
304 		}
305 		if (this.configuration == null) {
306 			throw new NullPointerException("Null networking configuration");
307 		}
308 		if (this.commandFactory == null) {
309 			throw new NullPointerException("Null command factory");
310 		}
311 		if (this.weights != null && this.servers == null) {
312 			throw new NullPointerException("Empty server list");
313 		}
314 	}
