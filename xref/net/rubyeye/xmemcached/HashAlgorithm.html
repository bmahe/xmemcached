11  /**
12   * Known hashing algorithms for locating a server for a key. Note that all hash
13   * algorithms return 64-bits of hash, but only the lower 32-bits are
14   * significant. This allows a positive 32-bit number to be returned for all
15   * cases.
16   */
17  public enum HashAlgorithm {
18  
19  	/**
20  	 * Native hash (String.hashCode()).
21  	 */
22  	NATIVE_HASH,
23  	/**
24  	 * CRC32_HASH as used by the perl API. This will be more consistent both
25  	 * across multiple API users as well as java versions, but is mostly likely
26  	 * significantly slower.
27  	 */
28  	CRC32_HASH,
29  	/**
30  	 * FNV hashes are designed to be fast while maintaining a low collision
31  	 * rate. The FNV speed allows one to quickly hash lots of data while
32  	 * maintaining a reasonable collision rate.
33  	 * 
34  	 * @see http://www.isthe.com/chongo/tech/comp/fnv/
35  	 * @see http://en.wikipedia.org/wiki/Fowler_Noll_Vo_hash
36  	 */
37  	FNV1_64_HASH,
38  	/**
39  	 * Variation of FNV.
40  	 */
41  	FNV1A_64_HASH,
42  	/**
43  	 * 32-bit FNV1.
44  	 */
45  	FNV1_32_HASH,
46  	/**
47  	 * 32-bit FNV1a.
48  	 */
49  	FNV1A_32_HASH,
50  	/**

52  	 */
53  	KETAMA_HASH,

56  	 * From mysql source

58  	MYSQL_HASH,

67  	LUA_HASH,

82  	/**
83  	 * Compute the hash for the given key.
84  	 * 
85  	 * @return a positive integer hash
86  	 */
87  	public long hash(final String k) {
88  		long rv = 0;
89  		switch (this) {
90  		case NATIVE_HASH:
91  			rv = k.hashCode();
92  			break;
93  		case CRC32_HASH:
94  			// return (crc32(shift) >> 16) & 0x7fff;
95  			CRC32 crc32 = new CRC32();
96  			crc32.update(ByteUtils.getBytes(k));
97  			rv = crc32.getValue() >> 16 & 0x7fff;
98  			break;
99  		case FNV1_64_HASH: {
100 			// Thanks to pierre@demartines.com for the pointer
101 			rv = FNV_64_INIT;
102 			int len = k.length();
103 			for (int i = 0; i < len; i++) {
104 				rv *= FNV_64_PRIME;
105 				rv ^= k.charAt(i);
106 			}
107 		}
108 			break;
109 		case FNV1A_64_HASH: {
110 			rv = FNV_64_INIT;
111 			int len = k.length();
112 			for (int i = 0; i < len; i++) {
113 				rv ^= k.charAt(i);
114 				rv *= FNV_64_PRIME;
115 			}
116 		}
117 			break;
118 		case FNV1_32_HASH: {
119 			rv = FNV_32_INIT;
120 			int len = k.length();
121 			for (int i = 0; i < len; i++) {
122 				rv *= FNV_32_PRIME;
123 				rv ^= k.charAt(i);
124 			}
125 		}
126 			break;
127 		case FNV1A_32_HASH: {
128 			rv = FNV_32_INIT;
129 			int len = k.length();
130 			for (int i = 0; i < len; i++) {
131 				rv ^= k.charAt(i);
132 				rv *= FNV_32_PRIME;
133 			}
134 		}
135 			break;
136 		case ELECTION_HASH:
137 		case KETAMA_HASH:
138 			byte[] bKey = computeMd5(k);
139 			rv = (long) (bKey[3] & 0xFF) << 24 | (long) (bKey[2] & 0xFF) << 16
140 					| (long) (bKey[1] & 0xFF) << 8 | bKey[0] & 0xFF;
141 			break;
142 
143 		case MYSQL_HASH:
144 			int nr2 = 4;
145 			for (int i = 0; i < k.length(); i++) {
146 				rv ^= ((rv & 63) + nr2) * k.charAt(i) + (rv << 8);
147 				nr2 += 3;
148 			}
149 			break;
150 		case ELF_HASH:
151 			long x = 0;
152 			for (int i = 0; i < k.length(); i++) {
153 				rv = (rv << 4) + k.charAt(i);
154 				if ((x = rv & 0xF0000000L) != 0) {
155 					rv ^= x >> 24;
156 					rv &= ~x;
157 				}
158 			}
159 			rv = rv & 0x7FFFFFFF;
160 			break;
161 		case RS_HASH:
162 			long b = 378551;
163 			long a = 63689;
164 			for (int i = 0; i < k.length(); i++) {
165 				rv = rv * a + k.charAt(i);
166 				a *= b;
167 			}
168 			rv = rv & 0x7FFFFFFF;
169 			break;
170 		case LUA_HASH:
171 			int step = (k.length() >> 5) + 1;
172 			rv = k.length();
173 			for (int len = k.length(); len >= step; len -= step) {
174 				rv = rv ^ (rv << 5) + (rv >> 2) + k.charAt(len - 1);
175 			}
176 			break;
177 		case ONE_AT_A_TIME:
178 			try {
179 				int hash = 0;
180 				for (byte bt : k.getBytes("utf-8")) {
181 					hash += (bt & 0xFF);
182 					hash += (hash << 10);
183 					hash ^= (hash >>> 6);
184 				}
185 				hash += (hash << 3);
186 				hash ^= (hash >>> 11);
187 				hash += (hash << 15);
188 				rv = hash;
189 			} catch (UnsupportedEncodingException e) {
190 				throw new IllegalStateException("Hash function error", e);
191 			}
192 			break;
193 		default:
194 			assert false;
195 		}
196 
197 		return rv & 0xffffffffL; /* Convert to unsigned 32-bits */
198 	}

200 	private static ThreadLocal<MessageDigest> md5Local = new ThreadLocal<MessageDigest>();
201 
202 	/**
203 	 * Get the md5 of the given key.
204 	 */
205 	public static byte[] computeMd5(String k) {
206 		MessageDigest md5 = md5Local.get();
207 		if (md5 == null) {
208 			try {
209 				md5 = MessageDigest.getInstance("MD5");
210 				md5Local.set(md5);
211 			} catch (NoSuchAlgorithmException e) {
212 				throw new RuntimeException("MD5 not supported", e);
213 			}
214 		}
215 		md5.reset();
216 		md5.update(ByteUtils.getBytes(k));
217 		return md5.digest();
218 	}
