40  /**
41   * A {@link IoBuffer} that wraps a buffer and proxies any operations to it.
42   * <p>
43   * You can think this class like a {@link FilterOutputStream}. All operations
44   * are proxied by default so that you can extend this class and override
45   * existing operations selectively. You can introduce new operations, too.
46   * 
47   * @author The Apache MINA Project (dev@mina.apache.org)
48   * @version $Rev: 671827 $, $Date: 2008-06-26 10:49:48 +0200 (Thu, 26 Jun 2008)
49   *          $
50   */
51  public class IoBufferWrapper extends IoBuffer {
52  
53      /**
54       * The buffer proxied by this proxy.
55       */
56      private final IoBuffer buf;
57  
58  
59      /**
60       * Create a new instance.
61       * 
62       * @param buf
63       *            the buffer to be proxied
64       */
65      protected IoBufferWrapper(IoBuffer buf) {
66          if (buf == null) {
67              throw new NullPointerException("buf");
68          }
69          this.buf = buf;
70      }
71
