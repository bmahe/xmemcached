136 	@Override
137 	public void run() {
138 		selectorManager.notifyReady();
139 		while (selectorManager.isStarted() && selector.isOpen()) {
140 			try {
141 				beforeSelect();
142 				wakenUp.set(false);
143 				long before = -1;
144 				// Wether to look jvm bug
145 				if (SystemUtils.isLinuxPlatform()
146 						&& !SystemUtils.isAfterJava6u4Version()) {
147 					before = System.currentTimeMillis();
148 				}
149 				long wait = DEFAULT_WAIT;
150 				if (nextTimeout > 0) {
151 					wait = nextTimeout;
152 				}
153 				int selected = selector.select(wait);
154 				if (selected == 0) {
155 					if (before != -1) {
156 						lookJVMBug(before, selected, wait);
157 					}
158 					selectTries++;
159 					// check tmeout and idle
160 					nextTimeout = checkSessionTimeout();
161 					continue;
162 				} else {
163 					selectTries = 0;
164 				}
165 
166 			} catch (ClosedSelectorException e) {
167 				break;
168 			} catch (IOException e) {
169 				log.error("Reactor select error", e);
170 				if (selector.isOpen()) {
171 					continue;
172 				} else {
173 					break;
174 				}
175 			}
176 			Set<SelectionKey> selectedKeys = selector.selectedKeys();
177 			gate.lock();
178 			try {
179 				postSelect(selectedKeys, selector.keys());
180 				dispatchEvent(selectedKeys);
181 			} finally {
182 				gate.unlock();
183 			}
184 		}
185 		if (selector != null) {
186 			if (selector.isOpen()) {
187 				try {
188 					controller.closeChannel(selector);
189 					selector.close();
190 				} catch (IOException e) {
191 					controller.notifyException(e);
192 					log.error("stop reactor error", e);
193 				}
194 			}
195 		}

204 	 * @param wait

208 	private boolean lookJVMBug(long before, int selected, long wait)
209 			throws IOException {
210 		boolean seeing = false;
211 		long now = System.currentTimeMillis();
212 
213 		if (JVMBUG_THRESHHOLD > 0 && selected == 0 && wait > JVMBUG_THRESHHOLD
214 				&& now - before < wait / 4 && !wakenUp.get() /* waken up */
215 				&& !Thread.currentThread().isInterrupted()/* Interrupted */) {
216 			jvmBug.incrementAndGet();
217 			if (jvmBug.get() >= JVMBUG_THRESHHOLD2) {
218 				gate.lock();
219 				try {
220 					lastJVMBug = now;
221 					log.warn("JVM bug occured at "
222 							+ new Date(lastJVMBug)
223 							+ ",http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=6403933,reactIndex="
224 							+ reactorIndex);
225 					if (jvmBug1) {
226 						log.debug("seeing JVM BUG(s) - recreating selector,reactIndex="
227 								+ reactorIndex);
228 					} else {
229 						jvmBug1 = true;
230 						log.info("seeing JVM BUG(s) - recreating selector,reactIndex="
231 								+ reactorIndex);
232 					}
233 					seeing = true;
234 					final Selector new_selector = SystemUtils.openSelector();
235 
236 					for (SelectionKey k : selector.keys()) {
237 						if (!k.isValid() || k.interestOps() == 0) {
238 							continue;
239 						}
240 
241 						final SelectableChannel channel = k.channel();
242 						final Object attachment = k.attachment();
243 
244 						channel.register(new_selector, k.interestOps(),
245 								attachment);
246 					}
247 
248 					selector.close();
249 					selector = new_selector;
250 
251 				} finally {
252 					gate.unlock();
253 				}
254 				jvmBug.set(0);
255 
256 			} else if (jvmBug.get() == JVMBUG_THRESHHOLD
257 					|| jvmBug.get() == JVMBUG_THRESHHOLD1) {
258 				if (jvmBug0) {
259 					log.debug("seeing JVM BUG(s) - cancelling interestOps==0,reactIndex="
260 							+ reactorIndex);
261 				} else {
262 					jvmBug0 = true;
263 					log.info("seeing JVM BUG(s) - cancelling interestOps==0,reactIndex="
264 							+ reactorIndex);
265 				}
266 				gate.lock();
267 				seeing = true;
268 				try {
269 					for (SelectionKey k : selector.keys()) {
270 						if (k.isValid() && k.interestOps() == 0) {
271 							k.cancel();
272 						}
273 					}
274 				} finally {
275 					gate.unlock();
276 				}
277 			}
278 		} else {
279 			jvmBug.set(0);
280 		}
281 		return seeing;
282 	}

285 	 * Dispatch selected event
286 	 * 
287 	 * @param selectedKeySet
288 	 */
289 	public final void dispatchEvent(Set<SelectionKey> selectedKeySet) {
290 		Iterator<SelectionKey> it = selectedKeySet.iterator();
291 		boolean skipOpRead = false;
292 		while (it.hasNext()) {
293 			SelectionKey key = it.next();
294 			it.remove();
295 			if (!key.isValid()) {
296 				if (key.attachment() != null) {
297 					controller.closeSelectionKey(key);
298 				} else {
299 					key.cancel();
300 				}
301 				continue;
302 			}
303 			try {
304 				if (key.isValid() && key.isAcceptable()) {
305 					controller.onAccept(key);
306 					continue;
307 				}
308 				if (key.isValid()
309 						&& (key.readyOps() & SelectionKey.OP_WRITE) == SelectionKey.OP_WRITE) {
310 					// Remove write interest
311 					key.interestOps(key.interestOps() & ~SelectionKey.OP_WRITE);
312 					controller.onWrite(key);
313 					if (!controller.isHandleReadWriteConcurrently()) {
314 						skipOpRead = true;
315 					}
316 				}
317 				if (!skipOpRead
318 						&& key.isValid()
319 						&& (key.readyOps() & SelectionKey.OP_READ) == SelectionKey.OP_READ) {
320 					key.interestOps(key.interestOps() & ~SelectionKey.OP_READ);
321 					if (!controller.getStatistics().isReceiveOverFlow()) {
322 						// Remove read interest
323 						controller.onRead(key);
324 					} else {
325 						key.interestOps(key.interestOps()
326 								| SelectionKey.OP_READ);
327 					}
328 
329 				}
330 				if ((key.readyOps() & SelectionKey.OP_CONNECT) == SelectionKey.OP_CONNECT) {
331 					controller.onConnect(key);
332 				}
333 
334 			} catch (CancelledKeyException e) {
335 				// ignore
336 			} catch (RejectedExecutionException e) {
337 
338 				if (key.attachment() instanceof AbstractNioSession) {
339 					((AbstractNioSession) key.attachment()).onException(e);
340 				}
341 				controller.notifyException(e);
342 				if (selector.isOpen()) {
343 					continue;
344 				} else {
345 					break;
346 				}
347 			} catch (Exception e) {
348 				if (key.attachment() instanceof AbstractNioSession) {
349 					((AbstractNioSession) key.attachment()).onException(e);
350 				}
351 				controller.closeSelectionKey(key);
352 				controller.notifyException(e);
353 				log.error("Reactor dispatch events error", e);
354 				if (selector.isOpen()) {
355 					continue;
356 				} else {
357 					break;
358 				}
359 			}
360 		}
361 	}

381 	private final long checkSessionTimeout() {
382 		long nextTimeout = 0;
383 		if (configuration.getCheckSessionTimeoutInterval() > 0) {
384 			gate.lock();
385 			try {
386 				if (selectTries * 1000 >= configuration
387 						.getCheckSessionTimeoutInterval()) {
388 					nextTimeout = configuration
389 							.getCheckSessionTimeoutInterval();
390 					for (SelectionKey key : selector.keys()) {
391 
392 						if (key.attachment() != null) {
393 							long n = checkExpiredIdle(key,
394 									getSessionFromAttchment(key));
395 							nextTimeout = n < nextTimeout ? n : nextTimeout;
396 						}
397 					}
398 					selectTries = 0;
399 				}
400 			} finally {
401 				gate.unlock();
402 			}
403 		}
404 		return nextTimeout;
405 	}

434 		RegisterEvent event = null;
435 		while ((event = register.poll()) != null) {
436 			if (event.session != null) {

438 			} else {

448 	private final void dispatchSessionEvent(Session session, EventType event) {
449 		if (session.isClosed() && event != EventType.UNREGISTER) {
450 			return;
451 		}
452 		if (EventType.REGISTER.equals(event)) {
453 			controller.registerSession(session);
454 		} else if (EventType.UNREGISTER.equals(event)) {
455 			controller.unregisterSession(session);
456 		} else {
457 			((NioSession) session).onEvent(event, selector);
458 		}
459 	}

461 	public final void postSelect(Set<SelectionKey> selectedKeys,
462 			Set<SelectionKey> allKeys) {
463 		if (controller.getSessionTimeout() > 0
464 				|| controller.getSessionIdleTimeout() > 0) {
465 			for (SelectionKey key : allKeys) {
466 
467 				if (!selectedKeys.contains(key)) {
468 					if (key.attachment() != null) {
469 						checkExpiredIdle(key, getSessionFromAttchment(key));
470 					}
471 				}
472 			}
473 		}
474 	}

476 	private long checkExpiredIdle(SelectionKey key, Session session) {
477 		if (session == null) {
478 			return 0;
479 		}
480 		long nextTimeout = 0;
481 		boolean expired = false;
482 		if (controller.getSessionTimeout() > 0) {
483 			expired = checkExpired(key, session);
484 			nextTimeout = controller.getSessionTimeout();
485 		}
486 		if (controller.getSessionIdleTimeout() > 0 && !expired) {
487 			checkIdle(session);
488 			nextTimeout = controller.getSessionIdleTimeout();
489 		}
490 		return nextTimeout;
491 	}

493 	private final void checkIdle(Session session) {
494 		if (controller.getSessionIdleTimeout() > 0) {
495 			if (session.isIdle()) {
496 				((NioSession) session).onEvent(EventType.IDLE, selector);
497 			}
498 		}
499 	}

501 	private final boolean checkExpired(SelectionKey key, Session session) {
502 		if (session != null && session.isExpired()) {
503 			((NioSession) session).onEvent(EventType.EXPIRED, selector);
504 			controller.closeSelectionKey(key);
505 			return true;
506 		}
507 		return false;
508 	}
