54  	public static final String JAVA_VERSION = System
55  			.getProperty("java.version");
56  	private static boolean isAfterJava6u4Version = false;
57  	static {
58  		if (JAVA_VERSION != null) {
59  			// java4 or java5
60  			if (JAVA_VERSION.indexOf("1.4.") >= 0
61  					|| JAVA_VERSION.indexOf("1.5.") >= 0) {
62  				isAfterJava6u4Version = false;
63  			} else if (JAVA_VERSION.indexOf("1.6.") >= 0) {
64  				int index = JAVA_VERSION.indexOf("_");
65  				if (index > 0) {
66  					String subVersionStr = JAVA_VERSION.substring(index + 1);
67  					if (subVersionStr != null && subVersionStr.length() > 0) {
68  						try {
69  							int subVersion = Integer.parseInt(subVersionStr);
70  							if (subVersion >= 4) {
71  								isAfterJava6u4Version = true;
72  							}
73  						} catch (NumberFormatException e) {
74  
75  						}
76  					}
77  				}
78  				// after java6
79  			} else {
80  				isAfterJava6u4Version = true;
81  			}
82  		}
83  	}
84

105 	public static final Selector openSelector() throws IOException {
106 		Selector result = null;
107 		// check if it is linux os
108 		if (SystemUtils.isLinuxPlatform()) {
109 			try {
110 				Class<?> providerClazz = Class
111 						.forName("sun.nio.ch.EPollSelectorProvider");
112 				if (providerClazz != null) {
113 					try {
114 						Method method = providerClazz.getMethod("provider");
115 						if (method != null) {
116 							SelectorProvider selectorProvider = (SelectorProvider) method
117 									.invoke(null);
118 							if (selectorProvider != null) {
119 								result = selectorProvider.openSelector();
120 							}
121 						}
122 					} catch (Exception e) {
123 						// ignore
124 					}
125 				}
126 			} catch (Exception e) {
127 				// ignore
128 			}
129 		}
130 		if (result == null) {
131 			result = Selector.open();
132 		}
133 		return result;
134 
135 	}
