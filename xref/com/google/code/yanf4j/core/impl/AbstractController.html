335 	public synchronized void start() throws IOException {
336 		if (isStarted()) {
337 			return;
338 		}
339 		if (getHandler() == null) {
340 			throw new IOException("The handler is null");
341 		}
342 		if (getCodecFactory() == null) {
343 			setCodecFactory(new ByteBufferCodecFactory());
344 		}
345 		setStarted(true);
346 		setReadEventDispatcher(DispatcherFactory.newDispatcher(
347 				getReadThreadCount(),
348 				new ThreadPoolExecutor.CallerRunsPolicy(),
349 				"xmemcached-read-thread"));
350 		setWriteEventDispatcher(DispatcherFactory.newDispatcher(
351 				getWriteThreadCount(),
352 				new ThreadPoolExecutor.CallerRunsPolicy(),
353 				"xmemcached-write-thread"));
354 		setDispatchMessageDispatcher(DispatcherFactory.newDispatcher(
355 				getDispatchMessageThreadCount(),
356 				new ThreadPoolExecutor.CallerRunsPolicy(),
357 				"xmemcached-dispatch-thread"));
358 		startStatistics();
359 		start0();
360 		notifyStarted();
361 		shutdownHookThread = new Thread() {
362 			@Override
363 			public void run() {
364 				try {
365 					isHutdownHookCalled = true;
366 					AbstractController.this.stop();
367 				} catch (IOException e) {
368 					log.error("Stop controller fail", e);
369 				}
370 			}
371 		};
372 		Runtime.getRuntime().addShutdownHook(shutdownHookThread);
373 		log.warn("The Controller started at " + localSocketAddress + " ...");
374 	}

462 	public void stop() throws IOException {
463 		synchronized (this) {
464 			if (!isStarted()) {
465 				return;
466 			}
467 			setStarted(false);
468 		}
469 		for (Session session : sessionSet) {
470 			session.close();
471 		}
472 		stopStatistics();
473 		stopDispatcher();
474 		sessionSet.clear();
475 		notifyStopped();
476 		clearStateListeners();
477 		stop0();
478 		if (!isHutdownHookCalled) {
479 			Runtime.getRuntime().removeShutdownHook(shutdownHookThread);
480 		}
481 		log.info("Controller has been stopped.");
482 	}
