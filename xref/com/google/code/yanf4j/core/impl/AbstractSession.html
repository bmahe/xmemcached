232 	protected void dispatchReceivedMessage(final Object message) {
233 		if (dispatchMessageDispatcher == null) {
234 			long start = -1;
235 			if (statistics != null && statistics.isStatistics()) {
236 				start = System.currentTimeMillis();
237 			}
238 			onMessage(message, this);
239 			if (start != -1) {
240 				statistics
241 						.statisticsProcess(System.currentTimeMillis() - start);
242 			}
243 		} else {
244 
245 			dispatchMessageDispatcher.dispatch(new Runnable() {
246 				public void run() {
247 					long start = -1;
248 					if (statistics != null && statistics.isStatistics()) {
249 						start = System.currentTimeMillis();
250 					}
251 					onMessage(message, AbstractSession.this);
252 					if (start != -1) {
253 						statistics.statisticsProcess(System.currentTimeMillis()
254 								- start);
255 					}
256 				}
257 
258 			});
259 		}

279 	public void close() {
280 		synchronized (this) {
281 			if (isClosed()) {
282 				return;
283 			}
284 			setClosed(true);
285 		}
286 		try {
287 			closeChannel();
288 			clearAttributes();
289 			log.debug("session closed");
290 		} catch (IOException e) {
291 			onException(e);
292 			log.error("Close session error", e);
293 		} finally {
294 			onClosed();
295 		}
296 	}

379 		if (closed) {
