39  package com.google.code.yanf4j.core.impl;
40  
41  import java.util.concurrent.CancellationException;
42  import java.util.concurrent.CountDownLatch;
43  import java.util.concurrent.ExecutionException;
44  import java.util.concurrent.Future;
45  import java.util.concurrent.TimeUnit;
46  import java.util.concurrent.TimeoutException;
47  import java.util.concurrent.locks.ReentrantLock;
48  
49  
50  /**
51   * Simple {@link Future} implementation, which uses {@link ReentrantLock} to
52   * synchronize during the lifecycle.
53   * 
54   * @see Future
55   * @see ReentrantLock
56   * 
57   * @author Alexey Stashok
58   */
59  public class FutureLockImpl<R> implements Future<R> {
60  
61      private final ReentrantLock lock;
62  
63      private boolean isDone;
64  
65      private CountDownLatch latch;
66  
67      private boolean isCancelled;
68      private Throwable failure;
69  
70      protected R result;
71  
72  
73      public FutureLockImpl() {
74          this(new ReentrantLock());
75      }
76  
77  
78      public FutureLockImpl(ReentrantLock lock) {
79          this.lock = lock;
80          latch = new CountDownLatch(1);
81      }
82  
83  
84      /**
85       * Get current result value without any blocking.
86       * 
87       * @return current result value without any blocking.
88       */
89      public R getResult() {
90          try {
91              lock.lock();
92              return result;
93          }
94          finally {
95              lock.unlock();
96          }
97      }
98  
99  
100     /**
101      * Set the result value and notify about operation completion.
102      * 
103      * @param result
104      *            the result value
105      */
106     public void setResult(R result) {
107         try {
108             lock.lock();
109             this.result = result;
110             notifyHaveResult();
111         }
112         finally {
113             lock.unlock();
114         }
115     }
116 
117 
118     /**
119      * {@inheritDoc}
120      */
121     public boolean cancel(boolean mayInterruptIfRunning) {
122         try {
123             lock.lock();
124             isCancelled = true;
125             notifyHaveResult();
126             return true;
127         }
128         finally {
129             lock.unlock();
130         }
131     }
132 
133 
134     /**
135      * {@inheritDoc}
136      */
137     public boolean isCancelled() {
138         try {
139             lock.lock();
140             return isCancelled;
141         }
142         finally {
143             lock.unlock();
144         }
145     }
146 
147 
148     /**
149      * {@inheritDoc}
150      */
151     public boolean isDone() {
152         try {
153             lock.lock();
154             return isDone;
155         }
156         finally {
157             lock.unlock();
158         }
159     }
160 
161 
162     /**
163      * {@inheritDoc}
164      */
165     public R get() throws InterruptedException, ExecutionException {
166         latch.await();
167 
168         try {
169             lock.lock();
170             if (isCancelled) {
171                 throw new CancellationException();
172             }
173             else if (failure != null) {
174                 throw new ExecutionException(failure);
175             }
176 
177             return result;
178         }
179         finally {
180             lock.unlock();
181         }
182     }
183 
184 
185     /**
186      * {@inheritDoc}
187      */
188     public R get(long timeout, TimeUnit unit) throws InterruptedException, ExecutionException, TimeoutException {
189         boolean isTimeOut = !latch.await(timeout, unit);
190         try {
191             lock.lock();
192             if (!isTimeOut) {
193                 if (isCancelled) {
194                     throw new CancellationException();
195                 }
196                 else if (failure != null) {
197                     throw new ExecutionException(failure);
198                 }
199 
200                 return result;
201             }
202             else {
203                 throw new TimeoutException();
204             }
205         }
206         finally {
207             lock.unlock();
208         }
209     }
210 
211 
212     /**
213      * Notify about the failure, occured during asynchronous operation
214      * execution.
215      * 
216      * @param failure
217      */
218     public void failure(Throwable failure) {
219         try {
220             lock.lock();
221             this.failure = failure;
222             notifyHaveResult();
223         }
224         finally {
225             lock.unlock();
226         }
227     }
228 
229 
230     /**
231      * Notify blocked listeners threads about operation completion.
232      */
233     protected void notifyHaveResult() {
234         isDone = true;
235         latch.countDown();
236     }
237 }
